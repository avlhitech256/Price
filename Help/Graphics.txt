        // Проверка расширения:
        private static bool CheckFile(string file)
        {
            string check2="";
            int index = file.LastIndexOf('.') + 1;
            string check = file.Substring(index, file.Length - index).ToLower();
            if (check != "ini")
            {
                Bitmap bmp = new Bitmap(file);
                check2 = bmp.Height.ToString() + bmp.Width.ToString();
            }
            
            foreach (string expansion in expansions)
                if ((check == expansion)&&(check2=="10801920"))
                    return true;
            return false;
        }
